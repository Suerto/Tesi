% 1. TERMINI DI GLOSSARIO (Le definizioni estese)
% ==========================================
\newglossaryentry{wcag_ext}{
    name={Web Content Accessibility Guidelines (WCAG)},
    description={Oltre a definire standard etici e normativi, l'adozione delle WCAG impatta profondamente l'architettura \textit{frontend}. Impone l'uso di una semantica HTML rigorosa, la gestione avanzata del \textit{focus} via tastiera e l'integrazione di attributi ARIA, elementi che influenzano positivamente anche le prestazioni SEO degli applicativi.}
}

\newglossaryentry{uml_ext}{
    name={Unified Modeling Language (UML)},
    description={Nel contesto della moderna ingegneria del \textit{software} Agile, l'UML ha perso la connotazione di progetto rigido preliminare. Oggi viene impiegato prevalentemente secondo il paradigma dell'"\textit{UML as sketch}", fungendo da strumento di comunicazione rapida per illustrare \textit{pattern} architetturali e relazioni tra componenti critici.}
}
 
\newglossaryentry{ddd_ext}{
    name={Domain-Driven Design},
    text={DDD},
    description={Paradigma architetturale che affronta la complessità del \textit{software} introducendo il concetto di "Linguaggio Ubiquo" condiviso tra sviluppatori ed esperti di dominio. Isola i modelli all'interno di confini logici precisi (\textit{Bounded Contexts}), prevenendo la corruzione delle regole di \textit{business} in sistemi su larga scala.}
}
 
\newglossaryentry{project-management}{
    name={Project Management},
    description={Nelle moderne realtà IT, questa disciplina si discosta dai modelli predittivi tradizionali (come il \textit{Waterfall}) per abbracciare \textit{framework} adattivi. Il \textit{focus} gestionale si sposta dalla rigida aderenza al piano preventivo alla massimizzazione del valore continuo consegnato allo \textit{stakeholder}.}
}

\newglossaryentry{otp_ext}{
    name={One Time Password (OTP)},
    description={A livello crittografico e architetturale, l'implementazione si basa tipicamente su algoritmi standardizzati basati sul tempo (TOTP) o su eventi (HOTP). Rappresenta il pilastro dei moderni sistemi di autenticazione a più fattori per la mitigazione radicale del furto di identità e degli attacchi di tipo \textit{replay}.}
}

\newglossaryentry{jwt_ext}{
    name={JSON Web Token (JWT)},
    description={Composto da tre sezioni codificate in Base64, delega la verifica dell'autenticità alla crittografia anziché alla memoria del \textit{server}. Essendo \textit{stateless}, scala orizzontalmente in modo eccellente, ma introduce sfide architetturali complesse per l'invalidazione anticipata (\textit{revocation}) prima della naturale scadenza.}
}

\newglossaryentry{dry_ext}{
    name={Don't Repeat Yourself (DRY)},
    description={L'applicazione rigorosa del DRY non si limita al mero riuso del codice, ma si estende all'architettura dei dati. L'obiettivo è stabilire una singola fonte di verità (\textit{Single Source of Truth}), evitando che la logica di \textit{business} venga duplicata tra \textit{database}, \textit{backend} e \textit{client}.}
}

\newglossaryentry{project-management_ext}{
    name={Project Management},
    description={Disciplina che, se applicata allo sviluppo \textit{software}, richiede un bilanciamento continuo tra vincoli di tempo, costo e qualità. Le metodologie contemporanee privilegiano l'ispezione empirica e l'adattamento rapido rispetto alla pianificazione monolitica iniziale.}
}

\newglossaryentry{ci-cd_ext}{
    name={Continuous Integration / Continuous Delivery (CI/CD)},
    description={Pratica fondamentale dell'approccio DevOps che garantisce la rilevazione precoce dei difetti tramite l'esecuzione di \textit{test} automatizzati e analisi statica del codice a ogni \textit{commit}, riducendo drasticamente i cicli di \textit{feedback} tra lo sviluppo e la messa in produzione.}
}

\newglossaryentry{mvp_ext}{
    name={Minimum Viable Product (MVP)},
    description={Concetto cardine della metodologia \textit{Lean Startup}. Più che un prodotto incompleto, rappresenta un processo iterativo volto a massimizzare l'apprendimento validato sui bisogni reali degli utenti, minimizzando lo sforzo ingegneristico iniziale e il rischio di fallimento sul mercato.}
}

\newglossaryentry{restful_ext}{
    name={RESTful},
    description={Modello architetturale che sfrutta appieno i metodi semantici del protocollo HTTP. Si fonda su vincoli rigidi quali l'assenza di stato (\textit{statelessness}), la cacheabilità delle risposte e l'interfaccia uniforme, disaccoppiando radicalmente l'evoluzione del \textit{client} da quella del \textit{server}.}
}

\newglossaryentry{api_ext}{
    name={API},
    description={Oltre a fungere da ponte tecnico tra applicazioni, le API moderne rappresentano veri e propri contratti \textit{software}. L'esposizione di interfacce ben progettate permette alle aziende di strutturare architetture a microservizi e di monetizzare i propri \textit{asset} digitali (\textit{API Economy}).}
}

\newglossaryentry{b2b_ext}{
    name={Business-to-Business (B2B)},
    description={Le architetture progettate per scenari B2B differiscono profondamente da quelle B2C. Richiedono tipicamente protocolli di autenticazione federata, la gestione di moli di dati transazionali massicce e integrazioni su misura con i sistemi ERP preesistenti dei \textit{partner} commerciali.}
}

\newglossaryentry{it_ext}{
    name={Information Technology (IT)},
    description={Settore industriale che oggi rappresenta la leva strategica principale per la trasformazione digitale delle imprese, inglobando domini ingegneristici avanzati quali la \textit{cybersecurity} proattiva, l'analisi dei \textit{Big Data} e l'architettura del \textit{Cloud Computing}.}
}

\newglossaryentry{erp_ext}{
    name={Enterprise Resource Planning (ERP)},
    description={I sistemi ERP contemporanei superano la semplice contabilità, integrando moduli di analisi predittiva e appoggiandosi ad architetture \textit{cloud-native} a microservizi per garantire resilienza, isolamento dei processi e scalabilità dinamica su scala globale.}
}

\newglossaryentry{ocx_ext}{
    name={OLE Control Extension (OCX)},
    description={Tecnologia legata all'ecosistema COM di Microsoft. Sebbene oggi siano considerati \textit{legacy} e superati da \textit{framework} gestiti (come .NET), questi controlli restano una presenza radicata e architetturalmente complessa da manutenere in molti applicativi industriali storici.}
}

\newglossaryentry{orm_ext}{
    name={Object-Relational Mapping (ORM)},
    description={Strumento architetturale che risolve il disallineamento di paradigmi (\textit{impedance mismatch}) tra oggetti e \textit{database} relazionali. Tecniche avanzate includono il \textit{lazy loading}, il tracciamento delle mutazioni e la gestione della concorrenza ottimistica tramite \textit{token} di riga.}
}

\newglossaryentry{poc_ext}{
    name={Proof of Concept (PoC)},
    description={Si distingue da un prototipo poiché il suo ciclo di vita è effimero: una volta validata o invalidata la fattibilità tecnica di una soluzione, il codice del PoC viene tipicamente scartato, evitando che \textit{script} sperimentali diventino debito tecnico in produzione.}
}

\newglossaryentry{vcs_ext}{
    name={Version Control System (VCS)},
    description={Sistemi moderni come Git utilizzano grafi aciclici diretti e algoritmi di crittografia per garantire l'integrità del codice, permettendo ramificazioni rapide (\textit{branching}) e flussi di lavoro asincroni altamente parallelizzati tra i membri del \textit{team} di sviluppo.}
}

\newglossaryentry{llm_ext}{
    name={Large Language Model (LLM)},
    description={Modelli fondazionali basati su reti neurali \textit{Transformer} che stanno rivoluzionando i paradigmi di sviluppo \textit{software}, fornendo capacità avanzate di auto-completamento semantico del codice, \textit{refactoring} assistito e identificazione precoce di vulnerabilità logiche.}
}

\newglossaryentry{rnd_ext}{
    name={Research and Development (R\&D)},
    description={Nelle aziende IT rappresenta un motore per la sopravvivenza tecnologica. Include lo \textit{scouting} di nuovi \textit{framework}, l'analisi dei \textit{trend} architetturali e la sperimentazione per prevenire l'obsolescenza dei prodotti proprietari in un mercato altamente volatile.}
}

\newglossaryentry{system_design}{
    name={System Design},
    description={Processo olistico che richiede la continua valutazione di compromessi architetturali. I progettisti devono bilanciare i parametri del teorema CAP (Coerenza, Disponibilità, Tolleranza alle partizioni), scegliere strategie di scalabilità e delineare i criteri di partizionamento dei dati.}
}

\newglossaryentry{client-server}{
    name={Client-server},
    description={Paradigma evolutosi in architetture distribuite multilivello (\textit{n-tier}), all'interno delle quali le logiche di \textit{business}, la presentazione e la persistenza vengono allocate su \textit{cluster} indipendenti per garantire scalabilità orizzontale e alta affidabilità.}
}

\newglossaryentry{firewall}{
    name={Firewall},
    description={Nel contesto della sicurezza informatica moderna, l'isolamento perimetrale classico fornito dai \textit{firewall} sta cedendo il passo a modelli architetturali di tipo \textit{Zero Trust}, che richiedono un'autenticazione continua e crittografata indipendentemente dalla topologia di rete.}
}

\newglossaryentry{batch}{
    name={Batch},
    description={Nonostante l'affermazione delle architetture guidate dagli eventi (\textit{Event-Driven}), i processi \textit{batch} asincroni mantengono un ruolo cruciale, rimanendo lo standard per elaborazioni massive di \textit{machine learning} e reportistica ad altissima intensità di operazioni I/O.}
}

\newglossaryentry{machine-learning}{
    name={Machine Learning},
    description={Disciplina che ribalta l'approccio ingegneristico classico: anziché codificare algoritmi deterministici per ottenere risposte, si forniscono al calcolatore grandi \textit{dataset} affinché il modello inferisca autonomamente i pesi matematici e le regole latenti.}
}

\newglossaryentry{vmware}{
    name={VMware},
    description={Pioniere della virtualizzazione x86, le cui tecnologie basate su \textit{hypervisor bare-metal} hanno consentito il consolidamento dei \textit{datacenter}, gettando le basi concettuali per l'infrastruttura \textit{cloud} e i successivi sistemi di orchestrazione a \textit{container}.}
}

\newglossaryentry{rate-limiting}
{
  name={Rate Limiting},
  description={Algoritmo di protezione dell'infrastruttura di rete (implementato tramite \textit{pattern} come \textit{Token Bucket}). Nelle architetture distribuite richiede l'uso di \textit{cache} in memoria ultra-veloci (come Redis) per sincronizzare i contatori tra le istanze e mitigare gli attacchi DoS.}
}

\newglossaryentry{geomarketing}{
    name={Geomarketing},
    description={L'efficacia di questa branca analitica è stata amplificata dall'adozione massiva dei Sistemi Informativi Territoriali (GIS) e dall'analisi in tempo reale dei dati geospaziali provenienti dai dispositivi \textit{mobile}, consentendo strategie di \textit{micro-targeting} spaziale.}
}
 
\newglossaryentry{scrum}{
    name={Scrum},
    description={\textit{Framework} metodologico fondato sull'adattamento empirico dei processi. Il pilastro è la decentralizzazione del potere decisionale: attribuisce autonomia esecutiva al \textit{team} di sviluppo e responsabilizza il \textit{Product Owner} nella prioritizzazione del valore di \textit{business}.}
}

\newglossaryentry{code-coverage}{
    name={Code coverage},
    description={Sebbene sia un indicatore fondamentale della salute di una \textit{codebase}, un'elevata copertura non garantisce matematicamente l'assenza di \textit{bug} logici. L'ingegneria moderna suggerisce di affiancare tale metrica a tecniche di \textit{test} di mutazione e collaudi \textit{end-to-end}.}
}

\newglossaryentry{solid}{
    name={SOLID},
    description={Acronimo di cinque direttive di progettazione \textit{Object-Oriented} formalizzate da Robert C. Martin. Costituiscono la linea guida universale per ottenere architetture ad alta coesione e basso accoppiamento, prevenendo la fragilità strutturale del \textit{software}.}
}

\newglossaryentry{git}{
    name={Git},
    description={A differenza dei sistemi centralizzati del passato, memorizza la storia del codice sotto forma di istantanee compresse del \textit{filesystem}. L'uso di alberi di Merkle garantisce integrità crittografica, permettendo \textit{branching} rapido e risoluzioni asincrone dei conflitti di fusione.}
}

\newglossaryentry{refactoring}{
    name={Refactoring},
    description={Pratica ingegneristica che consiste nell'alterare la struttura interna di un componente senza modificarne il comportamento osservabile. È lo strumento principale per abbattere il debito tecnico accumulato e mantenere il codice estensibile nel lungo periodo.}
}

\newglossaryentry{azure}{
    name={Microsoft Azure},
    description={Piattaforma di \textit{cloud computing} la cui profonda integrazione nativa con il \textit{framework} .NET, con l'IDE Visual Studio e con i sistemi di \textit{Identity Management} (Active Directory) la posiziona come scelta architetturale predominante per i processi di modernizzazione \textit{enterprise}.}
}

\newglossaryentry{cloud}{
    name={Cloud Computing},
    description={Modello operativo basato sul concetto di \textit{utility computing}, all'interno del quale potenza di calcolo e archiviazione vengono allocati elasticamente e fatturati a consumo, eliminando la necessità di stime di dimensionamento (\textit{capacity planning}) \textit{hardware} massive.}
}

\newglossaryentry{dotnet}{
    name={.NET},
    description={Nato come \textit{framework} chiuso per l'ecosistema Windows, ha subito una radicale riscrittura strutturale, trasformandosi in un ambiente di esecuzione \textit{open source}, altamente performante e multipiattaforma, unificando lo sviluppo \textit{backend}, \textit{frontend} e \textit{mobile}.}
}

\newglossaryentry{entity-framework}{
    name={Entity Framework},
    description={ORM di riferimento nell'ecosistema .NET. Supporta architetturalmente l'approccio \textit{Code First}, in cui il \textit{database} viene modellato dalle classi di dominio in C\#, e l'approccio \textit{Database First}, in cui il codice sorgente viene ingegnerizzato inversamente da uno schema SQL.}
}

\newglossaryentry{embedded}{
    name={Embedded},
    description={L'iniezione di istruzioni SQL a basso livello direttamente nei linguaggi \textit{host} garantiva prestazioni massime nei sistemi \textit{legacy}. Oggi, tale accoppiamento rigido viene evitato a favore degli ORM per ragioni di sicurezza, modularità architetturale e facilità di \textit{testing} isolato.}
}
