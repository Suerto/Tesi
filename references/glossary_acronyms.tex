% ==========================================
% 1. TERMINI DI GLOSSARIO (Le definizioni estese)
% ==========================================
\newglossaryentry{ci-cd_ext}{
    name={Continuous Integration / Continuous Delivery (CI/CD)},
    description={Pratica fondamentale dell'approccio DevOps. Oltre all'automazione del rilascio, garantisce la rilevazione precoce dei difetti tramite test continui e analisi statica del codice, riducendo drasticamente i cicli di feedback tra la fase di sviluppo e la messa in produzione.}
}

\newglossaryentry{mvp_ext}{
    name={Minimum Viable Product (MVP)},
    description={Concetto cardine della metodologia Lean Startup ideata da Eric Ries. Più che un prodotto incompleto, rappresenta un processo iterativo volto a massimizzare l'apprendimento validato sui bisogni reali degli utenti, minimizzando lo sforzo ingegneristico iniziale e il rischio di mercato.}
}

\newglossaryentry{restful_ext}{
    name={RESTful},
    description={Modello architetturale teorizzato da Roy Fielding. Sfrutta appieno i metodi semantici del protocollo HTTP e si fonda su vincoli rigidi quali l'assenza di stato (statelessness), l'interfaccia uniforme e la separazione netta tra i domini di competenza di client e server.}
}

\newglossaryentry{api_ext}{
    name={API},
    description={Oltre a fungere da ponte tra applicazioni, le API moderne rappresentano veri e propri contratti software. L'esposizione di interfacce ben progettate permette alle aziende di strutturare architetture a microservizi e di monetizzare i propri asset digitali offrendoli a sviluppatori terzi.}
}

\newglossaryentry{b2b_ext}{
    name={Business-to-Business (B2B)},
    description={Nell'ambito dello sviluppo software, le architetture progettate per scenari B2B differiscono profondamente dalle soluzioni orientate al consumatore (B2C). Richiedono generalmente protocolli di autenticazione più rigorosi, gestione di moli di dati transazionali elevate e integrazioni su misura con i sistemi ERP preesistenti dei partner commerciali.}
}

\newglossaryentry{it_ext}{
    name={Information Technology (IT)},
    description={Settore industriale in continua evoluzione che oggi non si limita al solo supporto operativo, ma rappresenta la leva strategica principale per la trasformazione digitale delle imprese, inglobando domini avanzati come la cybersecurity, l'analisi dei Big Data e il cloud computing.}
}

\newglossaryentry{erp_ext}{
    name={Enterprise Resource Planning (ERP)},
    description={L'evoluzione moderna dei sistemi di pianificazione dei fabbisogni materiali (MRP). I sistemi ERP contemporanei superano la semplice contabilità, integrando moduli di analisi predittiva e architetture cloud-native per garantire resilienza e scalabilità su scala globale.}
}

\newglossaryentry{ocx_ext}{
    name={OLE Control Extension (OCX)},
    description={Tecnologia strettamente legata all'ecosistema COM (Component Object Model) di Microsoft, estremamente popolare negli anni Novanta. Sebbene ormai considerati legacy e superati dai moderni framework, questi controlli restano una presenza radicata in molti applicativi industriali storici ancora in produzione.}
}

\newglossaryentry{orm_ext}{
    name={Object-Relational Mapping (ORM)},
    description={Strumento architetturale che risolve il cosiddetto "impedance mismatch" tra il paradigma di programmazione orientato agli oggetti e quello relazionale dei database. Tecniche avanzate di ORM includono il lazy loading, per posticipare il caricamento di dati pesanti, e la gestione della concorrenza ottimistica.}
}

\newglossaryentry{poc_ext}{
    name={Proof of Concept (PoC)},
    description={Si distingue nettamente da un prototipo o da un MVP poiché non è destinato a giungere sul mercato. Il suo ciclo di vita è effimero: una volta validata o invalidata l'ipotesi tecnica iniziale, il codice del PoC viene tipicamente scartato e l'applicativo viene riscritto applicando rigidi criteri di ingegneria del software.}
}

\newglossaryentry{vcs_ext}{
    name={Version Control System (VCS)},
    description={L'adozione di un sistema di tracciamento rappresenta il primo gradino di maturità metodologica per un team di sviluppo. Strumenti moderni come Git hanno rivoluzionato questo campo introducendo il concetto di controllo distribuito, permettendo ramificazioni rapide e flussi di lavoro asincroni altamente parallelizzati.}
}

\newglossaryentry{llm_ext}{
    name={Large Language Model (LLM)},
    description={Modelli fondazionali addestrati su petabyte di dati sfruttando architetture a reti neurali di tipo Transformer. Oltre all'elaborazione del linguaggio naturale, stanno rivoluzionando l'ingegneria del software fornendo capacità avanzate di auto-completamento del codice, refactoring assistito e generazione automatica di test.}
}

\newglossaryentry{rnd_ext}{
    name={Research and Development (R\&D)},
    description={Nelle aziende IT, la divisione R\&D non rappresenta un'attività teorica isolata, bensì un motore per la competitività. Include lo scouting di nuove tecnologie, l'analisi dei trend di mercato e la sperimentazione su architetture innovative al fine di prevenire l'obsolescenza dei prodotti software proprietari.}
}

\newglossaryentry{system_design}{
    name={System Design},
    description={Processo olistico che richiede la continua valutazione di compromessi architetturali. I progettisti di sistemi complessi devono bilanciare parametri critici stabiliti dal teorema CAP (Coerenza, Disponibilità, Tolleranza alle partizioni), scegliere tra strategie di scalabilità orizzontale o verticale, e delineare i criteri di partizionamento dei dati.}
}

\newglossaryentry{client-server}{
    name={Client-server},
    description={Paradigma che ha storicamente soppiantato l'era dell'elaborazione centralizzata sui mainframe. Nelle applicazioni moderne si è evoluto in architetture multilivello ("n-tier"), all'interno delle quali le logiche di business, la presentazione delle interfacce e la persistenza dei dati vengono ulteriormente distribuite su cluster di macchine per garantire alta affidabilità.}
}

\newglossaryentry{firewall}{
    name={Firewall},
    description={Nel contesto contemporaneo della sicurezza informatica, l'isolamento perimetrale classico fornito storicamente dai firewall sta gradualmente cedendo il passo a modelli architetturali di tipo "Zero Trust", all'interno dei quali l'affidabilità di un dispositivo non viene mai assunta a priori, indipendentemente dalla sua posizione all'interno o all'esterno della rete aziendale.}
}

\newglossaryentry{batch}{
    name={Batch},
    description={Nonostante l'affermazione delle architetture di elaborazione streaming in tempo reale, i processi batch mantengono un ruolo cruciale nei sistemi bancari e gestionali industriali, rimanendo lo standard di fatto per operazioni quali la quadratura dei conti notturna, il calcolo delle buste paga e la generazione di reportistica consolidata ad altissima intensità di I/O.}
}

\newglossaryentry{machine-learning}{
    name={Machine Learning},
    description={Disciplina che ribalta l'approccio alla programmazione algoritmica classica: invece di codificare regole logiche fisse per elaborare dati e ottenere risposte, il paradigma prevede di fornire al calcolatore grandi quantità di dati ed esempi pratici affinché sia il sistema stesso a dedurre matematicamente le regole e le correlazioni intrinseche.}
}

\newglossaryentry{vmware}{
    name={VMware},
    description={Pioniere della virtualizzazione x86 commerciale, le cui tecnologie hanno permesso il consolidamento massivo dei datacenter fisici attraverso l'introduzione di hypervisor "bare-metal". Queste innovazioni architetturali hanno gettato le basi concettuali per l'avvento del cloud computing moderno e dei successivi sistemi di orchestrazione a container.}
}

\newglossaryentry{geomarketing}{
    name={Geomarketing},
    description={L'efficacia e la granularità di questa branca analitica sono state enormemente amplificate dall'adozione massiva dei sistemi informativi territoriali (GIS) e dall'analisi in tempo reale dei dati di geolocalizzazione provenienti dai dispositivi mobili degli utenti, consentendo strategie di micro-targeting spaziale estremamente precise.}
}
 
\newglossaryentry{scrum}{
    name={Scrum},
    description={Framework metodologico ideato da Ken Schwaber e Jeff Sutherland, fondato sull'ispezione continua dei processi e sul loro adattamento. Il pilastro portante è il concetto di "potere decisionale decentralizzato", che attribuisce elevata autonomia esecutiva al team di sviluppo e responsabilizza la figura del Product Owner nella massimizzazione del valore consegnato.}
}

\newglossaryentry{code-coverage}{
    name={Code coverage},
    description={Sebbene venga considerata un indicatore fondamentale della salute di una codebase, un'elevata percentuale di copertura del codice non garantisce matematicamente l'assenza di falle o bug logici. Le best practice dell'ingegneria del software suggeriscono infatti di affiancare tale metrica a tecniche di test di mutazione e a un rigoroso collaudo di sistema di tipo end-to-end.}
}

\newglossaryentry{solid}{
    name={SOLID},
    description={Insieme di cinque direttive di progettazione promosse da Robert C. Martin ("Uncle Bob") tese a guidare lo sviluppatore verso architetture ad alta coesione e basso accoppiamento. La mancata osservanza di questi principi si traduce tipicamente in codice rigido, fragile o difficile da isolare, generando i cosiddetti "code smells".}
}

\newglossaryentry{git}{
    name={Git},
    description={Ideato originariamente da Linus Torvalds per ottimizzare la gestione del kernel Linux. A differenza dei sistemi di versionamento centralizzati come SVN, Git memorizza le informazioni sotto forma di istantanee (snapshot) del filesystem locale, garantendo un'integrità crittografica assoluta del tracciamento tramite l'uso intensivo di algoritmi di hashing SHA-1.}
}

\newglossaryentry{refactoring}{
    name={Refactoring},
    description={Pratica ingegneristica formalizzata da Martin Fowler, che prevede l'applicazione metodica di micro-trasformazioni strutturali al codice. L'obiettivo primario di tali interventi è contrastare il cosiddetto "decadimento del software", facilitando le implementazioni future e riducendo il debito tecnico accumulato nelle precedenti iterazioni di sviluppo.}
}

\newglossaryentry{azure}{
    name={Microsoft Azure},
    description={Ecosistema che offre l'accesso a oltre duecento servizi cloud diversificati. La sua profonda integrazione nativa con l'ambiente di sviluppo .NET e l'IDE Visual Studio la posiziona come la scelta architetturale di riferimento per molte realtà enterprise storiche impegnate in percorsi di modernizzazione della propria infrastruttura IT.}
}

\newglossaryentry{cloud}{
    name={Cloud Computing},
    description={Modello operativo basato sul concetto di "utility computing", all'interno del quale le risorse informatiche vengono erogate e fatturate in base all'effettivo consumo. Le architetture cloud si ramificano tipicamente in tre paradigmi fondamentali di astrazione dei servizi: IaaS (Infrastructure as a Service), PaaS (Platform as a Service) e SaaS (Software as a Service).}
}

\newglossaryentry{dotnet}{
    name={.NET},
    description={Nato originariamente come framework chiuso, vincolato esclusivamente all'ecosistema Windows, .NET ha subito nel corso degli anni una radicale metamorfosi open source e multipiattaforma. Questo percorso ha permesso di unificare lo sviluppo di interfacce utente, servizi di backend e applicazioni mobile sotto l'ombrello di un singolo ambiente di esecuzione condiviso.}
}

\newglossaryentry{entity-framework}{
    name={Entity Framework},
    description={Nello sviluppo di applicativi .NET, questo ORM supporta operativamente due paradigmi principali di progettazione: l'approccio "Code First", in cui il database viene auto-generato direttamente a partire dalla stesura delle classi C\#, e l'approccio "Database First", all'interno del quale il codice sorgente viene ingegnerizzato inversamente a partire da uno schema relazionale preesistente.}
}

\newglossaryentry{embedded}{
    name={Embedded},
    description={L'iniezione di codice SQL a basso livello direttamente all'interno di linguaggi host procedurali (come il C o il C++) garantiva prestazioni di esecuzione critiche negli anni Ottanta e Novanta. Tuttavia, tale pratica oggi viene frequentemente scartata in favore degli ORM o delle stored procedure per motivazioni legate alla sicurezza, alla manutenibilità e alla testabilità del sistema.}
}
