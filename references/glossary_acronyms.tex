% ==========================================
% 1. TERMINI DI GLOSSARIO (Le definizioni estese)
% ==========================================
\newglossaryentry{crud_ext}{
    name={CRUD},
    description={Acronimo di \textit{Create, Read, Update, Delete}. Costituisce il paradigma fondazionale per la gestione della persistenza dei dati. Nelle architetture \textit{RESTful}, gli sviluppatori mappano tipicamente queste quattro operazioni sui verbi semantici del protocollo HTTP (POST, GET, PUT/PATCH, DELETE) per garantire interfacce uniformi e prevedibili.}
}

\newglossaryentry{cors_ext}{
    name={Cross-Origin Resource Sharing (CORS)},
    description={Protocollo di sicurezza basato su intestazioni HTTP che delega al \textit{server} il controllo degli accessi trans-origine. I \textit{browser} lo impiegano per mitigare gli attacchi di tipo \textit{Cross-Site Request Forgery} (CSRF) e bloccano le chiamate non autorizzate prima che il \textit{client} possa processare la risposta.}
}

\newglossaryentry{dto_ext}{
    name={Data Transfer Object (DTO)},
    description={Modello architetturale (\textit{design pattern}) che ottimizza le comunicazioni di rete. Un DTO funge da contenitore serializzabile, del tutto privo di logica di \textit{business}, che aggrega dati provenienti da entità molteplici. Il suo impiego riduce il numero di chiamate remote e nasconde la struttura interna del \textit{database} ai \textit{client} esterni.}
}

\newglossaryentry{captcha_ext}{
    name={CAPTCHA},
    description={Acronimo di \textit{Completely Automated Public Turing test to tell Computers and Humans Apart}. Misura di sicurezza crittografica e comportamentale che i portali \textit{web} impiegano per discriminare le interazioni umane dal traffico generato da \textit{bot}. I modelli più recenti analizzano schemi di navigazione invisibili per ridurre l'attrito durante l'autenticazione.}
}

\newglossaryentry{ux_ext}{
    name={User Experience (UX)},
    description={Disciplina ingegneristica e psicologica che modella l'interazione tra l'utente e il prodotto digitale. Supera la pura progettazione visiva (\textit{User Interface}) per abbracciare l'accessibilità, l'architettura dell'informazione e la fluidità dei percorsi cognitivi, con l'obiettivo di minimizzare la frustrazione operativa.}
}

\newglossaryentry{xss_ext}{
    name={Cross-Site Scripting (XSS)},
    description={Vulnerabilità di sicurezza tipica delle applicazioni \textit{web} che affligge il livello di presentazione. I malintenzionati sfruttano la mancata sanificazione degli \textit{input} per iniettare \textit{script} esecutivi nel \textit{browser} delle vittime, un'azione che compromette le sessioni attive e facilita l'esfiltrazione di \textit{token} di accesso.}
}

\newglossaryentry{http_ext}{
    name={HTTP},
    description={Acronimo di \textit{Hypertext Transfer Protocol}. Protocollo applicativo \textit{stateless} che regola la trasmissione dei documenti ipertestuali sulla rete Internet. L'evoluzione verso gli standard HTTP/2 e HTTP/3 ha introdotto il parallelismo delle richieste e la compressione delle intestazioni per abbattere la latenza nelle architetture moderne.}
}

\newglossaryentry{wcag_ext}{
    name={Web Content Accessibility Guidelines (WCAG)},
    description={Linee guida internazionali pubblicate dal W3C per l'accessibilità dei contenuti \textit{web}. La loro adozione impone l'impiego di HTML semantico, un contrasto cromatico elevato e il supporto alla sintesi vocale. Il rispetto di questi standard estende la fruibilità degli applicativi alle persone con disabilità visive, uditive o motorie.}
}

\newglossaryentry{uml_ext}{
    name={Unified Modeling Language (UML)},
    description={Linguaggio di modellazione visiva standardizzato per l'ingegneria del \textit{software}. Le metodologie \textit{Agile} contemporanee lo hanno spogliato della sua rigidità documentale per promuovere il paradigma dell'\textit{UML as sketch}: i \textit{team} lo impiegano oggi per tracciare diagrammi rapidi, utili a comunicare \textit{pattern} strutturali senza l'onere di un mantenimento formale continuo.}
}
 
\newglossaryentry{ddd_ext}{
    name={Domain-Driven Design},
    text={DDD},
    description={Paradigma ingegneristico che affronta la complessità del \textit{software} ancorando il codice al dominio applicativo. Introdotto da Eric Evans, impone un Linguaggio Ubiquo condiviso tra sviluppatori ed esperti di settore. La separazione in \textit{Bounded Contexts} isola le logiche di \textit{business} e previene l'entropia architetturale nei sistemi \textit{enterprise} distribuiti.}
}
 
\newglossaryentry{project-management}{
    name={Project Management},
    description={Disciplina organizzativa che orchestra risorse, tempi e costi per completare un progetto IT. I \textit{framework} moderni abbandonano il rigore predittivo dei modelli a cascata in favore di approcci empirici e adattivi, che massimizzano la consegna continua di valore e accorciano i cicli di \textit{feedback} con gli \textit{stakeholder}.}
}

\newglossaryentry{otp_ext}{
    name={One Time Password (OTP)},
    description={Acronimo di \textit{One Time Password}. Meccanismo crittografico per la generazione di credenziali monouso, spesso derivate tramite algoritmi basati sul tempo (TOTP) o su contatori sequenziali (HOTP). Rappresenta il cardine dell'autenticazione a più fattori, essenziale per neutralizzare le vulnerabilità delle \textit{password} statiche e gli attacchi di intercettazione.}
}

\newglossaryentry{jwt_ext}{
    name={JSON Web Token (JWT)},
    description={Standard aperto per la creazione di \textit{token} di accesso che asseriscono dichiarazioni in formato JSON. La firma crittografica garantisce l'integrità del carico utile, mentre l'assenza di stato (\textit{statelessness}) solleva il \textit{server} dal mantenimento delle sessioni in memoria. Tale architettura favorisce una scalabilità orizzontale estrema.}
}

\newglossaryentry{dry_ext}{
    name={Don't Repeat Yourself (DRY)},
    description={Acronimo di \textit{Don't Repeat Yourself}. Principio di progettazione che vieta la duplicazione delle logiche di \textit{business} e dei dati. L'architettura deve prevedere una singola fonte di verità (\textit{Single Source of Truth}) per ogni conoscenza di sistema, una pratica che abbatte i costi di manutenzione e previene le incongruenze durante gli aggiornamenti.}
}

\newglossaryentry{project-management_ext}{
    name={Project Management},
    description={Disciplina che, se applicata allo sviluppo \textit{software}, richiede un bilanciamento continuo tra vincoli di tempo, costo e qualità. Le metodologie contemporanee privilegiano l'ispezione empirica e l'adattamento rapido rispetto alla pianificazione monolitica iniziale.}
}

\newglossaryentry{ci-cd_ext}{
    name={Continuous Integration / Continuous Delivery (CI/CD)},
    description={Acronimo di \textit{Continuous Integration / Continuous Delivery}. Dorsale automatizzata della cultura DevOps che orchestra l'integrazione del codice. L'esecuzione sistematica di analisi statiche, \textit{test} unitari e procedure di rilascio a ogni \textit{commit} abbatte il rischio di regressioni e garantisce un codice costantemente pronto per la produzione.}
}

\newglossaryentry{mvp_ext}{
    name={Minimum Viable Product (MVP)},
    description={Acronimo di \textit{Minimum Viable Product}. Prototipo funzionale concepito per validare un'ipotesi di mercato con il minimo sforzo ingegneristico. A differenza di una dimostrazione puramente tecnica, l'MVP affronta un ciclo reale di utilizzo per raccogliere metriche empiriche e orientare gli sviluppi successivi verso i bisogni effettivi degli utenti.}
}

\newglossaryentry{restful_ext}{
    name={RESTful},
    description={Stile architetturale per i sistemi distribuiti che sfrutta pienamente i verbi e i codici di stato del protocollo HTTP. Impone vincoli strutturali netti, tra cui l'assenza di stato della sessione e l'interfaccia uniforme, per garantire il disaccoppiamento totale tra l'evoluzione del \textit{client} e i meccanismi di persistenza del \textit{server}.}
}

\newglossaryentry{api_ext}{
    name={API},
    description={Acronimo di \textit{Application Programming Interface}. Contratto formale che definisce le modalità di comunicazione tra componenti \textit{software} distinti. Le API moderne superano la pura funzione tecnica: costituiscono prodotti a sé stanti che espongono i servizi aziendali, favoriscono l'integrazione con sistemi di terze parti e alimentano l'economia dei microservizi.}
}

\newglossaryentry{b2b_ext}{
    name={Business-to-Business (B2B)},
    description={Acronimo di \textit{Business-to-Business}. Modello di transazione digitale tra imprese, caratterizzato da logiche di scambio diametralmente opposte a quelle per il consumatore finale. Tali architetture richiedono un'elevata automazione procedurale, l'esposizione di API di integrazione e l'adesione a severi standard di autenticazione federata.}
}

\newglossaryentry{it_ext}{
    name={Information Technology (IT)},
    description={Acronimo di \textit{Information Technology}. Macro-settore ingegneristico che progetta e gestisce l'infrastruttura tecnologica globale. Oltre al tradizionale supporto operativo, oggi l'IT ingloba la modernizzazione dei processi aziendali, l'ingegneria del \textit{software} distribuito e la protezione degli \textit{asset} informatici attraverso pratiche di \textit{cybersecurity}.}
}

\newglossaryentry{erp_ext}{
    name={Enterprise Resource Planning (ERP)},
    description={Acronimo di \textit{Enterprise Resource Planning}. Piattaforma centralizzata che governa i flussi logistici, finanziari e produttivi di un'azienda. Gli ERP contemporanei superano la pura archiviazione contabile: si affidano ad architetture \textit{cloud} per offrire moduli analitici intelligenti, interoperabilità avanzata e visibilità in tempo reale sui processi industriali.}
}

\newglossaryentry{ocx_ext}{
    name={OLE Control Extension (OCX)},
    description={Acronimo di \textit{OLE Control Extension}. Moduli \textit{software} precompilati basati sulla tecnologia Component Object Model (COM) di Microsoft. Benché l'industria odierna li classifichi come \textit{legacy}, numerosi ecosistemi gestionali storici vi fanno ancora affidamento per estendere le capacità delle interfacce grafiche in ambiente Windows.}
}

\newglossaryentry{orm_ext}{
    name={Object-Relational Mapping (ORM)},
    description={Acronimo di \textit{Object-Relational Mapping}. Livello di astrazione che risolve il disallineamento strutturale tra il paradigma orientato agli oggetti e le tabelle relazionali. Un ORM traduce fluidamente le istanze in \textit{query} SQL ottimizzate, e offre strumenti nativi per la gestione delle transazioni, la concorrenza ottimistica e il caricamento differito (\textit{lazy loading}) dei dati.}
}

\newglossaryentry{poc_ext}{
    name={Proof of Concept (PoC)},
    description={Acronimo di \textit{Proof of Concept}. Artefatto ingegneristico effimero che i \textit{team} realizzano per validare la fattibilità tecnica di una specifica tecnologia. A differenza del Minimum Viable Product, il PoC non raggiunge l'utente finale: l'azienda scarta il suo codice sperimentale una volta estratte le metriche necessarie, per evitare l'accumulo di debito tecnico.}
}

\newglossaryentry{vcs_ext}{
    name={Version Control System (VCS)},
    description={Acronimo di \textit{Version Control System}. Infrastruttura crittografica e relazionale che traccia l'evoluzione di una \textit{codebase}. I sistemi distribuiti contemporanei, come Git, memorizzano la cronologia in grafi aciclici diretti; tale struttura abilita la clonazione locale dei \textit{repository}, la ramificazione indipendente del lavoro e la risoluzione sicura dei conflitti.}
}

\newglossaryentry{llm_ext}{
    name={Large Language Model (LLM)},
    description={Acronimo di \textit{Large Language Model}. Rete neurale profonda addestrata su immensi \textit{corpus} testuali mediante l'architettura \textit{Transformer}. Nell'ingegneria del \textit{software}, gli LLM agiscono da assistenti cognitivi che deducono il contesto del progetto per generare frammenti di codice, automatizzare la documentazione e individuare vulnerabilità logiche.}
}

\newglossaryentry{rnd_ext}{
    name={Research and Development (R\&D)},
    description={Acronimo di \textit{Research and Development}. Dipartimento nevralgico che le aziende tecnologiche impiegano per scongiurare l'obsolescenza dei propri prodotti. Le sue attività includono l'esplorazione dei \textit{trend} architetturali emergenti e la prototipazione rapida di nuovi \textit{framework}, per garantire la competitività dell'impresa in un mercato ad alta volatilità.}
}

\newglossaryentry{system_design}{
    name={System Design},
    description={Disciplina ingegneristica che orchestra architetture su larga scala. I progettisti valutano continuamente i compromessi tra affidabilità, latenza e complessità, e prendono decisioni cruciali sul partizionamento dei \textit{database}, sui protocolli di comunicazione e sulle strategie di replica, per soddisfare i vincoli del teorema CAP.}
}

\newglossaryentry{client-server}{
    name={Client-server},
    description={Paradigma informatico che divide i carichi di elaborazione. Il \textit{server} centralizza la logica di \textit{business} e la protezione dei dati, mentre i \textit{client} periferici gestiscono la presentazione visiva e l'interazione con l'utente. Questa separazione strutturale facilita la manutenzione e la scalabilità orizzontale dei servizi.}
}

\newglossaryentry{firewall}{
    name={Firewall},
    description={Sistema di sicurezza perimetrale che ispeziona il traffico di rete in transito. Le implementazioni moderne di \textit{Next-Generation Firewall} (NGFW) superano il semplice filtro di porte e protocolli: analizzano profondamente il carico utile delle applicazioni e integrano meccanismi di prevenzione delle intrusioni per neutralizzare le minacce avanzate.}
}

\newglossaryentry{batch}{
    name={Batch},
    description={Paradigma di elaborazione asincrona in cui il sistema esegue blocchi sequenziali di operazioni senza richiedere l'interazione umana. Risulta imprescindibile per le elaborazioni ad alta intensità computazionale, come l'addestramento di modelli di \textit{machine learning}, le riconciliazioni bancarie notturne e la migrazione massiva di dati.}
}

\newglossaryentry{machine-learning}{
    name={Machine Learning},
    description={Sottocampo dell'intelligenza artificiale che sostituisce la programmazione esplicita con l'inferenza statistica. Gli algoritmi di apprendimento automatico assimilano schemi latenti a partire da vasti \textit{dataset}, e ottimizzano progressivamente i propri parametri matematici per formulare previsioni accurate su dati inediti.}
}

\newglossaryentry{vmware}{
    name={VMware},
    description={Società pionieristica che ha democratizzato la virtualizzazione delle architetture x86. Gli \textit{hypervisor bare-metal} dell'azienda consentono l'esecuzione simultanea di molteplici macchine virtuali isolate sul medesimo \textit{hardware}, una tecnologia che ha gettato le basi operative per il consolidamento dei \textit{datacenter} e il \textit{cloud computing} moderno.}
}

\newglossaryentry{rate-limiting}
{
  name={Rate Limiting},
  description={Meccanismo di difesa infrastrutturale che disciplina il volume di traffico indirizzato a una risorsa di rete. L'implementazione di algoritmi quali il \textit{Token Bucket} limita il tasso di richieste per singolo indirizzo IP o \textit{token} utente, un approccio vitale per prevenire gli attacchi di negazione del servizio (DoS) e garantire la stabilità del sistema.}
}

\newglossaryentry{geomarketing}{
    name={Geomarketing},
    description={Disciplina analitica che fonde le scienze commerciali con l'intelligenza spaziale. L'elaborazione in tempo reale di flussi geospaziali permette alle aziende di mappare i bacini di utenza, ottimizzare la logistica delle consegne e indirizzare campagne di acquisizione con estrema precisione territoriale.}
}
 
\newglossaryentry{scrum}{
    name={Scrum},
    description={\textit{Framework} metodologico per lo sviluppo \textit{Agile} dei prodotti. Organizza il lavoro in iterazioni brevi (\textit{Sprint}) e assegna responsabilità trasparenti a figure cardine: lo \textit{Scrum Master} rimuove gli impedimenti procedurali, mentre il \textit{Product Owner} massimizza il valore del prodotto e governa dinamicamente la lista delle priorità.}
}

\newglossaryentry{code-coverage}{
    name={Code coverage},
    description={Metrica quantitativa che esprime la percentuale di codice sorgente effettivamente attraversata dagli \textit{unit test} automatizzati. Benché l'industria la consideri uno standard di qualità, una copertura del cento per cento non rileva difetti logici o di progettazione, motivo per cui i \textit{team} esperti vi affiancano verifiche architetturali e collaudi di integrazione.}
}

\newglossaryentry{solid}{
    name={SOLID},
    description={Insieme di cinque postulati cardine (\textit{Single Responsibility, Open-Closed, Liskov Substitution, Interface Segregation, Dependency Inversion}) ideati per la programmazione orientata agli oggetti. La loro rigorosa applicazione contrasta la rigidità del \textit{software} e favorisce la creazione di architetture altamente modulari, estensibili e facili da sottoporre a \textit{test}.}
}

\newglossaryentry{git}{
    name={Git},
    description={Motore di controllo di versione distribuito che ha ridefinito la collaborazione nello sviluppo \textit{software}. Le sue fondamenta basate su istantanee crittografate garantiscono un tracciamento inalterabile della \textit{codebase}, mentre la gestione fluida dei \textit{branch} locali abilita flussi di lavoro asincroni per \textit{team} di dimensioni globali.}
}

\newglossaryentry{refactoring}{
    name={Refactoring},
    description={Intervento ingegneristico sistematico che purifica e ottimizza la struttura interna di un programma senza alterarne il comportamento esterno. Questa pratica disinnesca il progressivo deterioramento del codice (\textit{software rot}) e facilita l'inclusione di nuove funzionalità in sistemi altrimenti difficili da manutenere.}
}

\newglossaryentry{azure}{
    name={Microsoft Azure},
    description={Ecosistema di \textit{cloud computing} pubblico amministrato da Microsoft. Fornisce una vasta gamma di servizi distribuiti, dall'infrastruttura come servizio (IaaS) a modelli di calcolo \textit{serverless}. La sua eccellente integrazione nativa con l'infrastruttura di identità aziendale lo rende il fornitore privilegiato per le reti \textit{enterprise}.}
}

\newglossaryentry{cloud}{
    name={Cloud Computing},
    description={Paradigma informatico che distribuisce risorse computazionali (\textit{server}, \textit{database}, reti) su richiesta attraverso Internet. Solleva le imprese dall'onere di mantenere costosi centri dati proprietari, e offre un'elasticità illimitata abbinata a un modello economico puramente basato sull'effettivo consumo operativo.}
}

\newglossaryentry{dotnet}{
    name={.NET},
    description={Piattaforma di sviluppo unificata prodotta da Microsoft. Dopo l'abbandono dell'architettura proprietaria originaria, .NET si è affermato come un ecosistema \textit{open source} multipiattaforma ad alte prestazioni, che consente ai programmatori di impiegare la medesima infrastruttura per creare librerie \textit{web}, applicativi \textit{desktop} e servizi per il \textit{cloud}.}
}

\newglossaryentry{entity-framework}{
    name={Entity Framework},
    description={\textit{Framework} Object-Relational Mapper ufficiale per il linguaggio C\#. Elimina la necessità di stendere \textit{query} SQL dirette e fornisce un \textit{set} di strumenti per interagire con i dati tramite la forte tipizzazione del codice. I suoi meccanismi di migrazione del \textit{database} allineano automaticamente lo schema relazionale ai cambiamenti della logica di dominio.}
}

\newglossaryentry{embedded}{
    name={Embedded},
    description={Informatica integrata che vincola l'elaborazione a uno specifico contesto \textit{hardware} o \textit{software}. Nell'ambito dei \textit{database} storici, le istruzioni SQL integrate indicano l'inclusione di costrutti di interrogazione direttamente all'interno dei linguaggi imperativi nativi, un retaggio che l'ingegneria moderna sconsiglia per via della sua scarsa modularità.}
}

\newglossaryentry{bcrypt}{
    name={BCrypt},
    description={Algoritmo crittografico asimmetrico specializzato nell'archiviazione sicura delle \textit{password}. Deriva dal cifrario Blowfish e implementa un rimescolamento dinamico (\textit{salt}) unito a un fattore di costo (\textit{work factor}) adattivo. Tale configurazione contrasta le capacità di calcolo crescente delle moderne schede grafiche e frustra gli attacchi di tipo \textit{rainbow table}.}
}
